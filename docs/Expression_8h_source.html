104 template<L::ExpressionTraversalOrder order>
105 void L::Expression::depth_first_traversal(const std::function<void(Expression & expr)>& explore) {
106 
107  std::function<void(Expression&)> traversal;
108  traversal = [&traversal, &explore](Expression& current){
109  if constexpr(order == PostOrder) {
110  if (current.has_child(Left)) traversal(current.child(Left));
111  if (current.has_child(Right)) traversal(current.child(Right));
112  explore(current);
113  } else if constexpr(order == InOrder) {
114  if (current.has_child(Left)) traversal(current.child(Left));
115  explore(current);
116  if (current.has_child(Right)) traversal(current.child(Right));
117  } else {
118  explore(current);
119  if (current.has_child(Left)) traversal(current.child(Left));
120  if (current.has_child(Right)) traversal(current.child(Right));
121  }
122  };
123 
124  try { traversal(*this); } catch (const StopIteration& err) {}
125 }
