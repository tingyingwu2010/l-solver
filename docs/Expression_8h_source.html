306 void L::Expression::depth_first_traversal(const std::function<void(Expression & expr)>& explore) {
307 
308  std::function<void(Expression&)> traversal;
309  traversal = [&traversal, &explore](Expression& current){
310  if constexpr(order == PostOrder) {
311  if (current.has_child(Left)) traversal(current.child(Left));
312  if (current.has_child(Right)) traversal(current.child(Right));
313  explore(current);
314  } else if constexpr(order == InOrder) {
315  if (current.has_child(Left)) traversal(current.child(Left));
316  explore(current);
317  if (current.has_child(Right)) traversal(current.child(Right));
318  } else {
319  explore(current);
320  if (current.has_child(Left)) traversal(current.child(Left));
321  if (current.has_child(Right)) traversal(current.child(Right));
322  }
323  };
324 
325  try { traversal(*this); } catch (const StopIteration& err) {}
326 }
